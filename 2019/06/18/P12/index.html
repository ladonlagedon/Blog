JavaScript 学习笔记 第九篇 Break 和 Continue

Jun 18, 2019 
for (var i = 0; i < 10; i++) {
    console.log(i);

    if (i == 5) {
        break;
    }
}

//for循环嵌套
for (var x = 1; x <= 4; x++) {
    for (var y = 1; y <= 8; y++) {
        console.log(x, y);
    }
}

//当 y = 2 时，跳出循环，输出 x 与 y 的值。执行完成的有 1,1; 1,2; 2,1; 2,2; 3,1; 3,2; 4,1; 4,2;
for (var x = 1; x <= 4; x++) {
    for (var y = 1; y <= 8; y++) {
        console.log(x, y);
        if (y == 2) {
            break;
        }
    }
}
// y = 2 时，输出 x 与 y 的值。
for (var x = 1; x <= 4; x++) {
    for (var y = 1; y <= 8; y++) {
        if (y == 2) {
            console.log(x, y);
        }
    }
}

//使用自定义外层名称，同时跳出内层与外层循环。Break 结束了整个循环。
Outside:
for (var x = 1; x <= 4; x++) {
    for (var y = 1; y <= 8; y++) {
        console.log(x, y);
        if (y == 2) {
            break Outside;
        }
    }
}

//当 a = 6 时，使用 continue 跳出本次循环，继续执行下一次循环，不能将 log 写在 continue 前面。
for (var a = 1; a <= 8; a++) {
    if (a == 6) {
        continue;
    }
    console.log(a);
}

// continue 仍然可运用在 for 循环嵌套中。
for (var x = 1; x <= 4; x++) {
    for (var y = 1; y <= 8; y++) {
        if (y == 5) {
            continue;
        }
        console.log(x, y);
    }
}

 // 使用自定义外层名称，并同时跳出内层与外层。由于是 continue 只跳出了本次循环，所以继续执行。
 Outside:
 for (var x = 1; x <= 4; x++) {
    for (var y = 1; y <= 8; y++) {
        if (y == 2) {
            continue Outside;
        }
        console.log(x, y);
    }
}
