JavaScript 学习笔记 第十篇 while 和 do-while

Jun 18, 2019 
// for(var a = 0;a < 10; a++){
//     console.log(a);
// }

// 使用 while 循环语句，实现和上方 for 循环相同的结果。执行顺序是先判断后执行。
var a = 0;
while(a < 10){
    console.log('While' + a); // 不满足条件时，后面的语句不会执行。 
    a++;
}
// 使用 do-while 语句实现和 while 语句相同的结果。这里的执行顺序是先执行后判断
var a = 0;
do{
    console.log('Do While' + a); // 不满足条件时，后面的语句仍然会执行一次。 
    a++;
}while(a < 10)

//死循环
while(true){

} //条件只用填上 true 即可

for(;true;){

} //条件前后仍需填上分号，避免语法错误。
