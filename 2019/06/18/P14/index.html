JavaScript 学习笔记 第十一篇 函数

Jun 18, 2019 
//函数的定义、调用。
function haha(){
    alert('Suprise');
}

function gg(){
    alert('Jesuis');
}

// 自己写的关于弹出窗口调用函数的例子
var a = parseInt(prompt('请输入你的号码','12'));

if(a == 10){
    haha();
}else{
    gg();
}
function sum(a,b){ // a,b 为形式参数 作为函数的形参隐式 var
    console.log(a,b);
    console.log(a+b);
    // arguements 既能存放参数，也能修改参数。示例将 11 传递给的参数 b 的值，修改为了100。
    arguments[1] = 100; 
    // arguments 存放了下面的实际参数，0,1 对应了下面实际参数的 10,11
    console.log(arguments[0],arguments[1]); 
    console.log(arguments.length); // arguments.length 表示了参数的个数。
}

sum(10,11); // 10,11 为实际参数。此处调用了 sum 方法，将 10,11 赋值给 a,b。
// sum(); // 如果不设置实际参数，则输出 undefined，结果为 NaN。
// sum(1,2,3); 如果设置两个参数个数不一样的函数，那么永远都只会执行更下面的函数，
// 没有设置的实际参数，则输出为 undefined。

function sum(a, b) {
    console.log(arguments);
    console.log(a + b);
}

// sum(10, 20, 30, 40, 50); // 设置多个参数时，根据形式参数个数调用，这里只会调用前两个。
// sum(11); // 设置的实际参数不够时，未定义的实际参数则会显示 undefided。结果为 NaN。

function sum() {
    // 0 个参数输出提示信息；一个参数，输出参数的平方，2个参数，输出两者之积；多于两个数出参数之和。
    if (arguments.length == 0) {
        console.log('没有参数');
    } else if (arguments.length == 1) {
        console.log(Math.pow(arguments[0], 2));
    } else if (arguments.length == 2) {
        console.log(arguments[0] * arguments[1]);
    } else {
        var sum = 0;
        for (var i = 0; i < arguments.length;i ++) {
            // sum = sum + arguments[i]; 
            // i定义了存放在 aguments 中参数的位置，即第几个（从零开始计算个数,第零个就是第一个）。
            // 参考下面的的函数
            sum += arguments[i];                   
        }                       
        console.log(sum);
    }
}

sum(); // 输出信息
sum(10); // 100
sum(10,20); // 200
sum(10,20,30); // 60
sum(10,20,30,40); // 100

function sum() {
    console.log(arguments[2]); // "[2]" 表示调用的是第三个数，因为从 0 开始计算，输出结果为 30。
}

sum(10, 20, 30);
