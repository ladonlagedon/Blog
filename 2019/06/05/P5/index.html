JavaScript 学习笔记 第二篇 类型转换

Jun 5, 2019 
//使用 parseInt进行类型转换(纯数字字符串，数字开头的字符串，只截取整数部分)
var age = prompt('请输入你的年龄',20);
var age2 = parseInt(age);
console.log(age,typeof age);
console.log(age2,typeof age2);

//在 console 中转换以及在 console 输出结果之前进行类型转换
var a = '12';
console.log(parseInt(a),typeof parseInt(a));

var b = '12';
var c = parseInt(b);
console.log(c,typeof c); 

//在 JavaScript中，数字加字符也能转换为纯数字类型，但条件是必须以数字开头。
var a = '18岁';
var b = parseInt(a);
console.log(b,typeof b); //输出为 18，类型为 number.
var a = '年龄18';
var b = parseInt(a);
console.log(b,typeof b); //输出为 NaN，类型为 number.

var a = '18.66';
var b = parseInt(a);
console.log(b,typeof b); //输出为 18，类型为 number.

//使用 parseFloat 进行类型转换(纯数字字符串，数字开头的字符串，只截取整数部分)
var a = '18.667';
var b = parseFloat(a);
console.log(b,typeof b); //输出为 18.667，类型为 number.

//同上，parseFloat 只截取数字部分。
var a = '18.667年';
var b = parseFloat(a);
console.log(b,typeof b); //输出为 18.667，类型为 number.

//同上，以非数字开头，输出结果为 NaN。
var a = '年龄18.667';
var b = parseFloat(a);
console.log(b,typeof b); //输出为 NaN，类型为 number.

//隐式转换，使用 '字符串减 0'，直接将字符串转换为 number 类型,但只适用于纯数字的情况。
var a = '18';
var b = a - 0;
console.log(b,typeof b);

var a = '18a';
var b = a - 0;
console.log(b,typeof b); //输出为 NaN，类型为 number.

//不能使用 '字符串 + 0'转换，结果仍然是字符串。
var a = '18';
var b = a + 0;
console.log(b,typeof b);

//使用 '数字 + 空字符串',直接将 number 转换为 string 类型。
var a = 11;
var b = a + '';
console.log(b,typeof b);
