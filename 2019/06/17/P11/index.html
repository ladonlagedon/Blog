JavaScript 学习笔记 第八篇 For 循环

Jun 17, 2019 
for(var i = 0;i <= 10;i++){
    alert(i);
}

// JavaScript 执行时，条件不满足时，会直接跳出，执行后面的语句。
for(var j = 0;j >= 10;j++){
    alert(j);
}

alert('After !!!');

for(var k = 0;k <= 14;k += 4){
    console.log(k);
}
for 循环判断 48 的约数,并计算对应约数的个数
var count = 0;
for (var m = 1; m <= 48; m++) {
    if (48 % m == 0) {
        console.log(m);
        count ++;
    }
}
console.log(count);
