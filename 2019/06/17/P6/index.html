JavaScript 学习笔记 第三篇 基本数学运算及隐式转换

Jun 17, 2019 
// JavaScript 中的隐式转换，只存在于非 '+' 的运算中，因为 '+' 既是连字符也是加号。
// 只要 undefined 参与运算 结果都是 NaN
// true false null 都会隐式转换 纯数字字符串会隐式转换(加法除外)

console.log(3 + 5); // 8
console.log(3 + '5'); // 35
console.log(3 - '5'); // -2
console.log(3 * '5'); // 15
console.log(3 / '5'); // 0.6

console.log('3' + '5'); // 35
console.log('3' - '5'); // -2
console.log('3' * '5'); // 15
console.log('3' / '5');

console.log(3 * true); // 3
console.log(3 / false); // Infinity
console.log(3 / null); // Infinity

console.log(10 > 8); //true
console.log(8 <= 8); //true
console.log(1 != 2); //true
console.log(1 === 1); //true

// 字符串和字符串之间比较的是字符编码顺序
console.log('a' > 'b'); //false
console.log('abc' < 'acd'); //true

// 纯数字会隐式转换 true 转为 1 false 转为 0 null不能和 0 判断相等
console.log(1 > '0'); //true
console.log(1 == true); //true
console.log(1 === true); //false
console.log(null > -0.00001); //true
console.log(null < 0.00001); //true

console.log(null == 0); //false

// NaN 不能和自己判断相等，也不能判断全等
console.log(NaN == NaN); //false
console.log(NaN === NaN); //false

console.log(3 > 2 > 1); //false 因为 '3 > 2' 为 true,实际上 '3 > 2 > 1' 是 'true > 1',所以结果为 false。
