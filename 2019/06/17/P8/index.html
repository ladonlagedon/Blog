JavaScript 学习笔记 第四篇 If-Else

Jun 17, 2019 
var score = parseInt(prompt('请输入成绩', 100));

// 编写 if - else 语句
if (score >= 60) {
    alert('及格了');
} else {
    alert('没及格');
}

//也可以只写 if 语句
if (score >= 60) {
    alert('及格了');
}

// if - else if 语句
if (score >= 90) {
    alert('真牛皮');
}else if(score >= 80 && score < 90){
    alert('算不错');
}else if(score >= 60 && score < 80){
    alert('还行吧');
}else if(score < 60){
    alert('你不行');
}

// 优化语句 因为前面已经判断过了，所以去掉多余的语句
if (score >= 90) {
    alert('真牛皮');
}else if(score >= 80){
    alert('算不错');
}else if(score >= 60){
    alert('还行吧');
}else if(score < 60){
    alert('你不行');
}

// 使用 if - else 语句 判断是否能结婚
var age = parseInt(prompt('请输入年龄', 100));
var sex = prompt('请输入性别', 100);

if(sex == '男'){
    if(age >= 22){
        alert('可以娶媳妇儿');
    }else{
        alert('不可以娶媳妇儿');
    }
}else{
    if(age >= 20){
        alert('可以嫁人');
    }else{
        alert('不可以嫁人'); 
    }
}
