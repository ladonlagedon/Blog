JavaScript 学习笔记 第四篇 逻辑运算与赋值运算

Jun 17, 2019 
// 使用 Javascript 进行逻辑运算。
true && alert('haha'); //会检查并执行后面的语句

false && alert('hahaa'); //不会检查并执行后面的语句

var age = parseInt(prompt('请输入年龄', 18));

age >= 18 && alert('满十八岁，可考驾照。');

age < 18 && alert('未满十八岁，不可考驾照。');

// 负性的：false,null,0,NaN,空字符串("")，undifined，不会执行后面的结果。
// 正性的：除了上面的，全是正性的。
0 && alert('haha');

false && alert('hah');

true && false || false && !true // false 运算顺序为 非、与、或。


// 赋值运算
// a += 1 ,事实上就是 a = a + 1;
var a = 10;
var b = a++; // a++ ,先赋值，再运算。
console.log(a, b); // 11,10

var a = 10;
var b = ++a; // ++a, 先运算，再赋值。
console.log(a, b); // 11,11


// a : 10 >> 11 >> 12
// a++ ++a
// b : 10 12
var a = 10;
var b = a++ + ++a;
console.log(a, b); // 12,22

// a : 10 >> 11 >> 12
// ++a a++
// b : 11 11
var a = 10;
var b = ++a + a++;
console.log(a, b); // 12,22

var a = 10;
var b = ++a + ++a;
console.log(a, b); // 12,23

var a = 10;
var b = a++ + a++;
console.log(a, b); // 12,21

var a = 10;
var b = a-- + a--;
console.log(a, b); // 8,19

// 运算符计算顺序：++ -- ！ 贴身的 >>> 数学 >>> 比较 >>> 逻辑 >>> 赋值
var a = true + false && !false >= 5 * 3 - false || true;
console.log(a); //true
