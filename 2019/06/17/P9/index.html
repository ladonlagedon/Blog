JavaScript 学习笔记 第五篇 Switch-Case

Jun 17, 2019 
var star = prompt('输入星座', '双鱼');

switch (star) {
    case '双鱼':
        alert('今天容易做事冲动不计后果');
        break;

    case '双子':
        alert('今天感情上你的情绪波动较大');
        break;

    case '水平':
        alert('今天个人状态一般，有些情绪不高');
        break;

    case '摩羯':
        alert('今天是时候做出点取舍，尽量离开那些让你感到不适的场所');
        break;

    default:
        alert('未收录');
        break;
}

// 如果当前 case 判断成功，会执行对应的代码，如果当前没写 break；会一直往后执行，遇见 case 不再判断，直到遇见 break 为止。

// 使用 switch case 判断月份对应天数。适用于有某些对应值的时候。
var month = parseInt(prompt('输入月份', 1));

// if一般使用 == 相等判断，switch 使用的是 === 全等判断。适用于有某些范围的时候。
if(month == 1){
    alert('31 Days');
}

switch (month) {
    case 1:
    case 3:
    case 5:
    case 7:
    case 8:
    case 10:
    case 12:
        alert('31天');
        break;

    case 4:
    case 6:
    case 9:
    case 11:
        alert('30天');
        break;

    case 2:
        alert('平年28天，闰年29天');
        break;

    default:
        alert('输入不正确');
        break;
}

//使用 switch case 判断范围成绩，避免了多个 case 的出现。
var score = parseInt(prompt('输入成绩', 1));

switch (true) {
    case score >= 90:
        alert('优秀');
        break;

    case score >= 80:
        alert('良好');
        break;

    case score >= 60:
        alert('及格');
        break;

    case score <= 60:
        alert('不及格');
        break;
}
