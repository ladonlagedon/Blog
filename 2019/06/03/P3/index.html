JavaScript 学习笔记 第一篇

Jun 3, 2019 
//输出警告框
alert("Hello,World!");

//单独定义变量和赋值
var a ;
a = 10;

var b;
b = 20;

//可单独定义变量和赋值，也可变量和赋值一起定义。
var c = 30;
//输出日志，算出结果
console.log(a,b);

//先输出再定义，会输出 undifined，因为 JavaScript中有变量声明提升的操作，
//会吧 c = 30；这句话提升到 console.log 前面，
//但 c = 10; 这句话的位置不变，所以相当于只定义了c，没有进行赋值，定义未完成，所以显示 undifined 。
console.log(c);
var c = 30;
//定义三个不同变量的值，并使用 typeof 输出变量的类型。
var a = 10,b = "20", c = false;
console.log(typeof a,typeof b,typeof c,typeof d);

//以下内容可得出 Javascript 中，任意类型的数字都默认为 number 类型。
var a = 10, b = 0101, c = 0xfa, d = 11.98, e = -26.75;
console.log(typeof a,typeof b,typeof c,typeof d,typeof e);

//2e6 代表了 2乘以十的 6次方。
var f = 2e6;
var g = 1.13124523e6;
var h = 241.23423e-3;

console.log(f,typeof f);
console.log(g,typeof g);
console.log(h,typeof h);

//不分正负 不分进制 不分整浮 Infinity NaN 都是 number类型。
var a = 1e99999;//Infinity
var b = -1e99999;//Infinity
var c = 1/2;//0.5
var d = 1/0;//Infinity
var e = 0/0;//NaN

console.log(a,typeof a);
console.log(b,typeof b);
console.log(c,typeof c);
console.log(d,typeof d);
console.log(e,typeof e);

//一些基本不会用上的运算
// Infinity + Infinity = Infinity
// Infinity - Infinity = NaN

// Infinity * Infinity = Infinity
// Infinity / Infinity = NaN

// Infinity - 1 = Infinity
// Infinity + 1 = Infinity

// - Infinity + Infinity = NaN
// Infinity + NaN = NaN

//通过引号包裹的皆输出为 String 类型。
var a = "haha", b = " ", c = "123", d = "";
console.log(typeof a,typeof b,typeof c,typeof d);

//未进行赋值的变量，值和类型,皆为 undifined。
var r;
console.log(r,typeof r);
