var score = parseInt(prompt('输入成绩',1));

var age = parseInt(prompt('请输入年龄', 100));

    age >= 22 ? alert('可以娶媳妇') : alert('不可以娶媳妇');
}else if(sex == '男'){
    age >= 20 ? alert('可以嫁人') : alert('不可以嫁人');
}else{
    alert('请输入正确的性别 !!!');
}

// 使用更为简单的方法判断适龄结婚条件
var age = parseInt(prompt('请输入年龄', 100));
var sex = prompt('请输入性别', '男');

var minage = sex == '男' ? 22 : 20; // 判断性别，并将对应的最小结婚年龄赋值给 minage
age >= minage ? alert('可以结婚') : alert('不可以结婚'); // 将用户输入的年龄和对应的最小结婚年龄对比。

JavaScript 学习笔记 第五篇 Switch-Case

Jun 17, 2019 
var star = prompt('输入星座', '双鱼');

switch (star) {
    case '双鱼':
        alert('今天容易做事冲动不计后果');
        break;

    case '双子':
        alert('今天感情上你的情绪波动较大');
        break;

    case '水平':
        alert('今天个人状态一般，有些情绪不高');
        break;

    case '摩羯':
        alert('今天是时候做出点取舍，尽量离开那些让你感到不适的场所');
        break;

    default:
        alert('未收录');
        break;
}

// 如果当前 case 判断成功，会执行对应的代码，如果当前没写 break；会一直往后执行，遇见 case 不再判断，直到遇见 break 为止。

// 使用 switch case 判断月份对应天数。适用于有某些对应值的时候。
var month = parseInt(prompt('输入月份', 1));

// if一般使用 == 相等判断，switch 使用的是 === 全等判断。适用于有某些范围的时候。
if(month == 1){
    alert('31 Days');
}

switch (month) {
    case 1:
    case 3:
    case 5:
    case 7:
    case 8:
    case 10:
    case 12:
        alert('31天');
        break;

    case 4:
    case 6:
    case 9:
    case 11:
        alert('30天');
        break;

    case 2:
        alert('平年28天，闰年29天');
        break;

    default:
        alert('输入不正确');
        break;
}

//使用 switch case 判断范围成绩，避免了多个 case 的出现。
var score = parseInt(prompt('输入成绩', 1));

switch (true) {
    case score >= 90:
        alert('优秀');
        break;

    case score >= 80:
        alert('良好');
        break;

    case score >= 60:
        alert('及格');
        break;

    case score <= 60:
        alert('不及格');
        break;
}

// 使用 Javascript 进行逻辑运算。
true && alert('haha'); //会检查并执行后面的语句

false && alert('hahaa'); //不会检查并执行后面的语句

age >= 18 && alert('满十八岁，可考驾照。');

age < 18 && alert('未满十八岁，不可考驾照。');

// 负性的：false,null,0,NaN,空字符串("")，undifined，不会执行后面的结果。
// 正性的：除了上面的，全是正性的。
0 && alert('haha');

false && alert('hah');

true && false || false && !true // false 运算顺序为 非、与、或。


// 赋值运算
// a += 1 ,事实上就是 a = a + 1;
var a = 10;
var b = a++; // a++ ,先赋值，再运算。
console.log(a, b); // 11,10

var a = 10;
var b = ++a; // ++a, 先运算，再赋值。
console.log(a, b); // 11,11


// a : 10 >> 11 >> 12
// a++ ++a
// b : 10 12
var a = 10;
var b = a++ + ++a;
console.log(a, b); // 12,22

// a : 10 >> 11 >> 12
// ++a a++
// b : 11 11
var a = 10;
var b = ++a + a++;
console.log(a, b); // 12,22

var a = 10;
var b = ++a + ++a;
console.log(a, b); // 12,23

var a = 10;
var b = a++ + a++;
console.log(a, b); // 12,21

var a = 10;
var b = a-- + a--;
console.log(a, b); // 8,19

// 运算符计算顺序：++ -- ！ 贴身的 >>> 数学 >>> 比较 >>> 逻辑 >>> 赋值
var a = true + false && !false >= 5 * 3 - false || true;
console.log(a); //true

JavaScript 学习笔记 第三篇 基本数学运算及隐式转换

Jun 17, 2019 
// JavaScript 中的隐式转换，只存在于非 '+' 的运算中，因为 '+' 既是连字符也是加号。
// 只要 undefined 参与运算 结果都是 NaN
// true false null 都会隐式转换 纯数字字符串会隐式转换(加法除外)

console.log(3 + 5); // 8
console.log(3 + '5'); // 35
console.log(3 - '5'); // -2
console.log(3 * '5'); // 15
console.log(3 / '5'); // 0.6

console.log('3' + '5'); // 35
console.log('3' - '5'); // -2
console.log('3' * '5'); // 15
console.log('3' / '5');

console.log(3 * true); // 3
console.log(3 / false); // Infinity
console.log(3 / null); // Infinity

console.log(10 > 8); //true
console.log(8 <= 8); //true
console.log(1 != 2); //true
console.log(1 === 1); //true

// 字符串和字符串之间比较的是字符编码顺序
console.log('a' > 'b'); //false
console.log('abc' < 'acd'); //true

// 纯数字会隐式转换 true 转为 1 false 转为 0 null不能和 0 判断相等
console.log(1 > '0'); //true
console.log(1 == true); //true
console.log(1 === true); //false
console.log(null > -0.00001); //true
console.log(null < 0.00001); //true

console.log(null == 0); //false

// NaN 不能和自己判断相等，也不能判断全等
console.log(NaN == NaN); //false
console.log(NaN === NaN); //false

console.log(3 > 2 > 1); //false 因为 '3 > 2' 为 true,实际上 '3 > 2 > 1' 是 'true > 1',所以结果为 false。
